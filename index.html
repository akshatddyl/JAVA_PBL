Database (PostgreSQL) — Ujjwal

tables:
  users       (id, username, email, password_hash, created_at)
  rooms       (id, name, invite_code [UUID], owner_id → users, created_at)
  room_members(room_id → rooms, user_id → users, joined_at)
  code_files  (id, room_id → rooms, filename, content TEXT, updated_at)
